39/**
40 * A keyboard that provides atomic typing actions.
41 *
42 * @constructor
43 * @param {{pressed: !Array.<!bot.Keyboard.Key>,
44            currentPos: number}=} opt_state Optional keyboard state.
45 * @extends {bot.Device}
46 */
47bot.Keyboard = function(opt_state) {
48  goog.base(this);
49
50  /** @private {boolean} */
51  this.editable_ = bot.dom.isEditable(this.getElement());
52
53  /** @private {number} */
54  this.currentPos_ = 0;
55
56  /** @private {!goog.structs.Set.<!bot.Keyboard.Key>} */
57  this.pressed_ = new goog.structs.Set();
58
59  if (opt_state) {
60    // If a state is passed, let's assume we were passed an object with
61    // the correct properties.
62    goog.array.forEach(opt_state['pressed'], function(key) {
63      this.setKeyPressed_(/** @type {!bot.Keyboard.Key} */ (key), true);
64    }, this);
65
66    this.currentPos_ = opt_state['currentPos'];
67  }
68};
69goog.inherits(bot.Keyboard, bot.Device);

72/**
73 * Maps characters to (key,boolean) pairs, where the key generates the
74 * character and the boolean is true when the shift must be pressed.
75 * @private {!Object.<string, {key: !bot.Keyboard.Key, shift: boolean}>}
76 * @const
77 */
78bot.Keyboard.CHAR_TO_KEY_ = {};

81/**
82 * Constructs a new key and, if it is a character key, adds a mapping from the
83 * character to is in the CHAR_TO_KEY_ map. Using this factory function instead
84 * of the new keyword, also helps reduce the size of the compiled Js fragment.
85 *
86 * @param {null|number|
87 *         {gecko: (?number), ieWebkit: (?number), opera: (?number)}} code
88 *     Either a single keycode or a record of per-browser keycodes.
89 * @param {string=} opt_char Character when shift is not pressed.
90 * @param {string=} opt_shiftChar Character when shift is pressed.
91 * @return {!bot.Keyboard.Key} The new key.
92 * @private
93 */
94bot.Keyboard.newKey_ = function(code, opt_char, opt_shiftChar) {
95  if (goog.isObject(code)) {
96    if (goog.userAgent.GECKO) {
97      code = code.gecko;
98    } else if (goog.userAgent.OPERA) {
99      code = code.opera;
100    } else {  // IE and Webkit
101      code = code.ieWebkit;
102    }
103  }
104  var key = new bot.Keyboard.Key(code, opt_char, opt_shiftChar);
105
106  // For a character key, potentially map the character to the key in the
107  // CHAR_TO_KEY_ map. Because of numpad, multiple keys may have the same
108  // character. To avoid mapping numpad keys, we overwrite a mapping only if
109  // the key has a distinct shift character.
110  if (opt_char && (!(opt_char in bot.Keyboard.CHAR_TO_KEY_) || opt_shiftChar)) {
111    bot.Keyboard.CHAR_TO_KEY_[opt_char] = {key: key, shift: false};
112    if (opt_shiftChar) {
113      bot.Keyboard.CHAR_TO_KEY_[opt_shiftChar] = {key: key, shift: true};
114    }
115  }
116
117  return key;
118};

122/**
123 * A key on the keyboard.
124 *
125 * @constructor
126 * @param {?number} code Keycode for the key; null for the (rare) case
127 *     that pressing the key issues no key events.
128 * @param {string=} opt_char Character when shift is not pressed; null
129 *     when the key does not cause a character to be typed.
130 * @param {string=} opt_shiftChar Character when shift is pressed; null
131 *     when the key does not cause a character to be typed.
132 */
133bot.Keyboard.Key = function(code, opt_char, opt_shiftChar) {
134  /** @type {?number} */
135  this.code = code;
136
137  /** @type {?string} */
138  this.character = opt_char || null;
139
140  /** @type {?string} */
141  this.shiftChar = opt_shiftChar || this.character;
142};

284/**
285 * Given a character, returns a pair of a key and a boolean: the key being one
286 * that types the character and the boolean indicating whether the key must be
287 * shifted to type it. This function will never return a numpad key; that is,
288 * it will always return a symbol key when given a number or math symbol.
289 *
290 * If given a character for which this module does not know the key (the key
291 * is not in the bot.Keyboard.Keys enumeration), returns a key that types the
292 * given character but has a (likely incorrect) keycode of zero.
293 *
294 * @param {string} ch Single character.
295 * @return {{key: !bot.Keyboard.Key, shift: boolean}} A pair of a key and
296 *     a boolean indicating whether shift must be pressed for the character.
297 */
298bot.Keyboard.Key.fromChar = function(ch) {
299  if (ch.length != 1) {
300    throw new bot.Error(bot.ErrorCode.UNKNOWN_ERROR,
301                        'Argument not a single character: ' + ch);
302  }
303  var keyShiftPair = bot.Keyboard.CHAR_TO_KEY_[ch];
304  if (!keyShiftPair) {
305    // We don't know the true keycode of non-US keyboard characters, but
306    // ch.toUpperCase().charCodeAt(0) should occasionally be right, and
307    // at least yield a positive number.
308    var upperCase = ch.toUpperCase();
309    var keyCode = upperCase.charCodeAt(0);
310    var key = bot.Keyboard.newKey_(keyCode, ch.toLowerCase(), upperCase);
311    keyShiftPair = {key: key, shift: (ch != key.character)};
312  }
313  return keyShiftPair;
314};

364/**
365 * Set the modifier state if the provided key is one, otherwise just add
366 * to the list of pressed keys.
367 * @param {!bot.Keyboard.Key} key The key to update.
368 * @param {boolean} isPressed Whether the key is pressed.
369 * @private
370 */
371bot.Keyboard.prototype.setKeyPressed_ = function(key, isPressed) {
372  if (goog.array.contains(bot.Keyboard.MODIFIERS, key)) {
373    var modifier = /** @type {bot.Device.Modifier}*/ (
374        bot.Keyboard.KEY_TO_MODIFIER_.get(key.code));
375    this.modifiersState.setPressed(modifier, isPressed);
376  }
377
378  if (isPressed) {
379    this.pressed_.add(key);
380  } else {
381    this.pressed_.remove(key);
382  }
383};

408/**
409 * Presses the given key on the keyboard. Keys that are pressed can be pressed
410 * again before releasing, to simulate repeated keys, except for modifier keys,
411 * which must be released before they can be pressed again.
412 *
413 * @param {!bot.Keyboard.Key} key Key to press.
414 */
415bot.Keyboard.prototype.pressKey = function(key) {
416  if (goog.array.contains(bot.Keyboard.MODIFIERS, key) && this.isPressed(key)) {
417    throw new bot.Error(bot.ErrorCode.UNKNOWN_ERROR,
418        'Cannot press a modifier key that is already pressed.');
419  }
420
421  // Note that GECKO is special-cased below because of
422  // https://bugzilla.mozilla.org/show_bug.cgi?id=501496. "preventDefault on
423  // keydown does not cancel following keypress"
424  var performDefault = !goog.isNull(key.code) &&
425      this.fireKeyEvent_(bot.events.EventType.KEYDOWN, key);
426
427  // Fires keydown and stops if unsuccessful.
428  if (performDefault || goog.userAgent.GECKO) {
429    // Fires keypress if required and stops if unsuccessful.
430    if (!this.requiresKeyPress_(key) ||
431        this.fireKeyEvent_(
432            bot.events.EventType.KEYPRESS, key, !performDefault)) {
433      if (performDefault) {
434        this.maybeSubmitForm_(key);
435        if (this.editable_) {
436          this.maybeEditText_(key);
437        }
438      }
439    }
440  }
441
442  this.setKeyPressed_(key, true);
443};

446/**
447 * Whether the given key currently requires a keypress.
448 * TODO: Make this dependent on the state of the modifier keys.
449 *
450 * @param {bot.Keyboard.Key} key Key.
451 * @return {boolean} Whether it requires a keypress event.
452 * @private
453 */
454bot.Keyboard.prototype.requiresKeyPress_ = function(key) {
455  if (key.character || key == bot.Keyboard.Keys.ENTER) {
456    return true;
457  } else if (goog.userAgent.WEBKIT) {
458    return false;
459  } else if (goog.userAgent.IE) {
460    return key == bot.Keyboard.Keys.ESC;
461  } else { // Gecko and Opera
462    switch (key) {
463      case bot.Keyboard.Keys.SHIFT:
464      case bot.Keyboard.Keys.CONTROL:
465      case bot.Keyboard.Keys.ALT:
466        return false;
467      case bot.Keyboard.Keys.META:
468      case bot.Keyboard.Keys.META_RIGHT:
469      case bot.Keyboard.Keys.CONTEXT_MENU:
470        return goog.userAgent.GECKO;
471      default:
472        return true;
473    }
474  }
475};

478/**
479 * Maybe submit a form if the ENTER key is released.  On non-FF browsers, firing
480 * the keyPress and keyRelease events for the ENTER key does not result in a
481 * form being submitted so we have to fire the form submit event as well.
482 *
483 * @param {bot.Keyboard.Key} key Key.
484 * @private
485 */
486bot.Keyboard.prototype.maybeSubmitForm_ = function(key) {
487  if (key != bot.Keyboard.Keys.ENTER) {
488    return;
489  }
490  if (goog.userAgent.GECKO ||
491      !bot.dom.isElement(this.getElement(), goog.dom.TagName.INPUT)) {
492    return;
493  }
494
495  var form = bot.Device.findAncestorForm(this.getElement());
496  if (form) {
497    var inputs = form.getElementsByTagName('input');
498    var hasSubmit = goog.array.some(inputs, function(e) {
499      return bot.Device.isFormSubmitElement(e);
500    });
501    // The second part of this if statement will always include forms on Safari
502    // version < 5.
503    if (hasSubmit || inputs.length == 1 ||
504        (goog.userAgent.WEBKIT && !bot.userAgent.isEngineVersion(534))) {
505      this.submitForm(form);
506    }
507  }
508};

511/**
512 * Maybe edit text when a key is pressed in an editable form.
513 *
514 * @param {!bot.Keyboard.Key} key Key that was pressed.
515 * @private
516 */
517bot.Keyboard.prototype.maybeEditText_ = function(key) {
518  if (key.character) {
519    this.updateOnCharacter_(key);
520  } else {
521    switch (key) {
522      case bot.Keyboard.Keys.ENTER:
523        this.updateOnEnter_();
524        break;
525      case bot.Keyboard.Keys.BACKSPACE:
526      case bot.Keyboard.Keys.DELETE:
527        this.updateOnBackspaceOrDelete_(key);
528        break;
529      case bot.Keyboard.Keys.LEFT:
530      case bot.Keyboard.Keys.RIGHT:
531        this.updateOnLeftOrRight_(key);
532        break;
533      case bot.Keyboard.Keys.HOME:
534      case bot.Keyboard.Keys.END:
535        this.updateOnHomeOrEnd_(key);
536        break;
537    }
538  }
539};

542/**
543 * Releases the given key on the keyboard. Releasing a key that is not
544 * pressed results in an exception.
545 *
546 * @param {!bot.Keyboard.Key} key Key to release.
547 */
548bot.Keyboard.prototype.releaseKey = function(key) {
549  if (!this.isPressed(key)) {
550    throw new bot.Error(bot.ErrorCode.UNKNOWN_ERROR,
551        'Cannot release a key that is not pressed. (' + key.code + ')');
552  }
553  if (!goog.isNull(key.code)) {
554    this.fireKeyEvent_(bot.events.EventType.KEYUP, key);
555  }
556
557  this.setKeyPressed_(key, false);
558};

561/**
562 * Given the current state of the SHIFT and CAPS_LOCK key, returns the
563 * character that will be typed is the specified key is pressed.
564 *
565 * @param {!bot.Keyboard.Key} key Key.
566 * @return {string} Character to be typed.
567 * @private
568 */
569bot.Keyboard.prototype.getChar_ = function(key) {
570  if (!key.character) {
571    throw new bot.Error(bot.ErrorCode.UNKNOWN_ERROR, 'not a character key');
572  }
573  var shiftPressed = this.isPressed(bot.Keyboard.Keys.SHIFT);
574  return /** @type {string} */ (shiftPressed ? key.shiftChar : key.character);
575};

588/**
589 * @param {!bot.Keyboard.Key} key Key with character to insert.
590 * @private
591 */
592bot.Keyboard.prototype.updateOnCharacter_ = function(key) {
593  if (bot.Keyboard.KEYPRESS_EDITS_TEXT_) {
594    return;
595  }
596
597  var character = this.getChar_(key);
598  var newPos = goog.dom.selection.getStart(this.getElement()) + 1;
599  goog.dom.selection.setText(this.getElement(), character);
600  goog.dom.selection.setStart(this.getElement(), newPos);
601  if (goog.userAgent.WEBKIT) {
602    this.fireHtmlEvent(bot.events.EventType.TEXTINPUT);
603  }
604  if (!bot.userAgent.IE_DOC_PRE9) {
605    this.fireHtmlEvent(bot.events.EventType.INPUT);
606  }
607  this.updateCurrentPos_(newPos);
608};

612bot.Keyboard.prototype.updateOnEnter_ = function() {
613  if (bot.Keyboard.KEYPRESS_EDITS_TEXT_) {
614    return;
615  }

617  // WebKit fires text input regardless of whether a new line is added, see:
618  // https://bugs.webkit.org/show_bug.cgi?id=54152
619  if (goog.userAgent.WEBKIT) {

635/**
636 * @param {!bot.Keyboard.Key} key Backspace or delete key.
637 * @private
638 */
639bot.Keyboard.prototype.updateOnBackspaceOrDelete_ = function(key) {
640  if (bot.Keyboard.KEYPRESS_EDITS_TEXT_) {
641    return;
642  }
643
644  // Determine what should be deleted.  If text is already selected, that
645  // text is deleted, else we move left/right from the current cursor.
646  var endpoints = goog.dom.selection.getEndPoints(this.getElement());
647  if (endpoints[0] == endpoints[1]) {
648    if (key == bot.Keyboard.Keys.BACKSPACE) {
649      goog.dom.selection.setStart(this.getElement(), endpoints[1] - 1);
650      // On IE, changing goog.dom.selection.setStart also changes the end.
651      goog.dom.selection.setEnd(this.getElement(), endpoints[1]);
652    } else {
653      goog.dom.selection.setEnd(this.getElement(), endpoints[1] + 1);
654    }
655  }
656
657  // If the endpoints are equal (e.g., the cursor was at the beginning/end
658  // of the input), the text field won't be changed.
659  endpoints = goog.dom.selection.getEndPoints(this.getElement());
660  var textChanged = !(endpoints[0] == this.getElement().value.length ||
661                      endpoints[1] == 0);
662  goog.dom.selection.setText(this.getElement(), '');
663
664  // Except for IE and GECKO, we need to fire the input event manually, but
665  // only if the text was actually changed.
666  // Note: Gecko has some strange behavior with the input event.  In a
667  //  textarea, backspace always sends an input event, while delete only
668  //  sends one if you actually change the text.
669  //  In a textbox/password box, backspace always sends an input event unless
670  //  the box has no text.  Delete behaves the same way in Firefox 3.0, but
671  //  in later versions it only fires an input event if no text changes.
672  if (!goog.userAgent.IE && textChanged ||
673      (goog.userAgent.GECKO && key == bot.Keyboard.Keys.BACKSPACE)) {
674    this.fireHtmlEvent(bot.events.EventType.INPUT);
675  }
676
677  // Update the cursor position
678  endpoints = goog.dom.selection.getEndPoints(this.getElement());
679  this.updateCurrentPos_(endpoints[1]);
680};

683/**
684 * @param {!bot.Keyboard.Key} key Special key to press.
685 * @private
686 */
687bot.Keyboard.prototype.updateOnLeftOrRight_ = function(key) {
688  var element = this.getElement();
689  var start = goog.dom.selection.getStart(element);
690  var end = goog.dom.selection.getEnd(element);
691
692  var newPos, startPos = 0, endPos = 0;
693  if (key == bot.Keyboard.Keys.LEFT) {
694    if (this.isPressed(bot.Keyboard.Keys.SHIFT)) {
695      // If the current position of the cursor is at the start of the
696      // selection, pressing left expands the selection one character to the
697      // left; otherwise, pressing left collapses it one character to the
698      // left.
699      if (this.currentPos_ == start) {
700        // Never attempt to move further left than the beginning of the text.
701        startPos = Math.max(start - 1, 0);
702        endPos = end;
703        newPos = startPos;
704      } else {
705        startPos = start;
706        endPos = end - 1;
707        newPos = endPos;
708      }
709    } else {
710      // With no current selection, pressing left moves the cursor one
711      // character to the left; with an existing selection, it collapses the
712      // selection to the beginning of the selection.
713      newPos = start == end ? Math.max(start - 1, 0) : start;
714    }
715  } else {  // (key == bot.Keyboard.Keys.RIGHT)
716    if (this.isPressed(bot.Keyboard.Keys.SHIFT)) {
717      // If the current position of the cursor is at the end of the selection,
718      // pressing right expands the selection one character to the right;
719      // otherwise, pressing right collapses it one character to the right.
720      if (this.currentPos_ == end) {
721        startPos = start;
722        // Never attempt to move further right than the end of the text.
723        endPos = Math.min(end + 1, element.value.length);
724        newPos = endPos;
725      } else {
726        startPos = start + 1;
727        endPos = end;
728        newPos = startPos;
729      }
730    } else {
731      // With no current selection, pressing right moves the cursor one
732      // character to the right; with an existing selection, it collapses the
733      // selection to the end of the selection.
734      newPos = start == end ? Math.min(end + 1, element.value.length) : end;
735    }
736  }
737
738  if (this.isPressed(bot.Keyboard.Keys.SHIFT)) {
739    goog.dom.selection.setStart(element, startPos);
740    // On IE, changing goog.dom.selection.setStart also changes the end.
741    goog.dom.selection.setEnd(element, endPos);
742  } else {
743    goog.dom.selection.setCursorPosition(element, newPos);
744  }
745  this.updateCurrentPos_(newPos);
746};

749/**
750 * @param {!bot.Keyboard.Key} key Special key to press.
751 * @private
752 */
753bot.Keyboard.prototype.updateOnHomeOrEnd_ = function(key) {
754  var element = this.getElement();
755  var start = goog.dom.selection.getStart(element);
756  var end = goog.dom.selection.getEnd(element);
757  // TODO: Handle multiline (TEXTAREA) elements.
758  if (key == bot.Keyboard.Keys.HOME) {
759    if (this.isPressed(bot.Keyboard.Keys.SHIFT)) {
760      goog.dom.selection.setStart(element, 0);
761      // If current position is at the end of the selection, typing home
762      // changes the selection to begin at the beginning of the text, running
763      // to the where the current selection begins.
764      var endPos = this.currentPos_ == start ? end : start;
765      // On IE, changing goog.dom.selection.setStart also changes the end.
766      goog.dom.selection.setEnd(element, endPos);
767    } else {
768      goog.dom.selection.setCursorPosition(element, 0);
769    }
770    this.updateCurrentPos_(0);
771  } else {  // (key == bot.Keyboard.Keys.END)
772    if (this.isPressed(bot.Keyboard.Keys.SHIFT)) {
773      if (this.currentPos_ == start) {
774        // Current position is at the beginning of the selection. Typing end
775        // changes the selection to begin where the current selection ends,
776        // running to the end of the text.
777        goog.dom.selection.setStart(element, end);
778      }
779      goog.dom.selection.setEnd(element, element.value.length);
780    } else {
781      goog.dom.selection.setCursorPosition(element, element.value.length);
782    }
783    this.updateCurrentPos_(element.value.length);
784  }
785};

797/**
798* @param {bot.events.EventType} type Event type.
799* @param {!bot.Keyboard.Key} key Key.
800* @param {boolean=} opt_preventDefault Whether the default event should be
801*     prevented. Defaults to false.
802* @return {boolean} Whether the event fired successfully or was cancelled.
803* @private
804*/
805bot.Keyboard.prototype.fireKeyEvent_ = function(type, key, opt_preventDefault) {
806  if (goog.isNull(key.code)) {
807    throw new bot.Error(bot.ErrorCode.UNKNOWN_ERROR,
808        'Key must have a keycode to be fired.');
809  }
810
811  var args = {
812    altKey: this.isPressed(bot.Keyboard.Keys.ALT),
813    ctrlKey: this.isPressed(bot.Keyboard.Keys.CONTROL),
814    metaKey: this.isPressed(bot.Keyboard.Keys.META),
815    shiftKey: this.isPressed(bot.Keyboard.Keys.SHIFT),
816    keyCode: key.code,
817    charCode: (key.character && type == bot.events.EventType.KEYPRESS) ?
818        this.getChar_(key).charCodeAt(0) : 0,
819    preventDefault: !!opt_preventDefault
820  };
821
822  return this.fireKeyboardEvent(type, args);
823};
