41/**
42 * Throws an exception if an element is not shown to the user, ignoring its
43 * opacity.
44
45 *
46 * @param {!Element} element The element to check.
47 * @see bot.dom.isShown.
48 * @private
49 */
50bot.action.checkShown_ = function(element) {
51  if (!bot.dom.isShown(element, /*ignoreOpacity=*/true)) {
52    throw new bot.Error(bot.ErrorCode.ELEMENT_NOT_VISIBLE,
53        'Element is not currently visible and may not be manipulated');
54  }
55};

58/**
59 * Throws an exception if the given element cannot be interacted with.
60 *
61 * @param {!Element} element The element to check.
62 * @throws {bot.Error} If the element cannot be interacted with.
63 * @see bot.dom.isInteractable.
64 * @private

66bot.action.checkInteractable_ = function(element) {
67  if (!bot.dom.isInteractable(element)) {

69        'Element is not currently interactable and may not be manipulated');
70
71  }

75/**
76 * Clears the given {@code element} if it is a editable text field.
77 *
78 * @param {!Element} element The element to clear.
79 * @throws {bot.Error} If the element is not an editable text field.
80 */
81bot.action.clear = function(element) {
82  bot.action.checkInteractable_(element);
83  if (!bot.dom.isEditable(element)) {
84    throw new bot.Error(bot.ErrorCode.INVALID_ELEMENT_STATE,
85        'Element must be user-editable in order to clear it.');
86  }
87
88  bot.action.LegacyDevice_.focusOnElement(element);
89  if (element.value) {
90    element.value = '';
91    bot.events.fire(element, bot.events.EventType.CHANGE);
92  }
93
94  if (bot.dom.isContentEditable(element)) {
95    // A single space is required, if you put empty string here you'll not be
96    // able to interact with this element anymore in Firefox.
97    element.innerHTML = ' ';
98    // contentEditable does not generate onchange event.
99  }
100};

114/**
115 * Types keys on the given {@code element} with a virtual keyboard.
116 *
117 * <p>Callers can pass in a string, a key in bot.Keyboard.Key, or an array
118 * of strings or keys. If a modifier key is provided, it is pressed but not
119 * released, until it is either is listed again or the function ends.
120 *
121 * <p>Example:
122 *   bot.keys.type(element, ['ab', bot.Keyboard.Key.LEFT,
123 *                           bot.Keyboard.Key.SHIFT, 'cd']);
124 *
125 * @param {!Element} element The element receiving the event.
126 * @param {(string|!bot.Keyboard.Key|!Array.<(string|!bot.Keyboard.Key)>)}
127 *    values Value or values to type on the element.
128 * @param {bot.Keyboard=} opt_keyboard Keyboard to use; if not provided,
129 *    constructs one.
130 * @param {boolean=} opt_persistModifiers Whether modifier keys should remain
131 *     pressed when this function ends.
132 * @throws {bot.Error} If the element cannot be interacted with.
133 */
134bot.action.type = function(
135    element, values, opt_keyboard, opt_persistModifiers) {
136  // If the element has already been brought into focus somewhow, typing is
137  // always allowed to proceed. Otherwise, we require the element be in an
138  // "interactable" state. For example, an element that is hidden by overflow
139  // can be typed on, so long as the user first tabs to it or the app calls
140  // focus() on the element first.
141  if (element != bot.dom.getActiveElement(element)) {
142    bot.action.checkInteractable_(element);
143    bot.action.scrollIntoView(element);
144  }
145
146  var keyboard = opt_keyboard || new bot.Keyboard();
147  keyboard.moveCursor(element);
148
149  function typeValue(value) {
150    if (goog.isString(value)) {
151      goog.array.forEach(value.split(''), function(ch) {
152        var keyShiftPair = bot.Keyboard.Key.fromChar(ch);
153        var shiftIsPressed = keyboard.isPressed(bot.Keyboard.Keys.SHIFT);
154        if (keyShiftPair.shift && !shiftIsPressed) {
155          keyboard.pressKey(bot.Keyboard.Keys.SHIFT);
156        }
157        keyboard.pressKey(keyShiftPair.key);
158        keyboard.releaseKey(keyShiftPair.key);
159        if (keyShiftPair.shift && !shiftIsPressed) {
160          keyboard.releaseKey(bot.Keyboard.Keys.SHIFT);
161        }
162      });
163    } else if (goog.array.contains(bot.Keyboard.MODIFIERS, value)) {
164      if (keyboard.isPressed(/** @type {!bot.Keyboard.Key} */ (value))) {
165        keyboard.releaseKey(value);
166      } else {
167        keyboard.pressKey(value);
168      }
169    } else {
170      keyboard.pressKey(value);
171      keyboard.releaseKey(value);
172    }
173  }
174
175  // mobile safari (iPhone / iPad). one cannot 'type' in a date field
176  // chrome implements this, but desktop Safari doesn't, what's webkit again?
177  if ((!(goog.userAgent.product.SAFARI && !goog.userAgent.MOBILE)) &&
178      goog.userAgent.WEBKIT && element.type == 'date') {
179    var val = goog.isArray(values)? values = values.join("") : values;
180    var datePattern = /\d{4}-\d{2}-\d{2}/;
181    if (val.match(datePattern)) {
182      // The following events get fired on iOS first
183      if (goog.userAgent.MOBILE && goog.userAgent.product.SAFARI) {
184        bot.events.fire(element, bot.events.EventType.TOUCHSTART);
185        bot.events.fire(element, bot.events.EventType.TOUCHEND);
186      }
187      bot.events.fire(element, bot.events.EventType.FOCUS);
188      element.value = val.match(datePattern)[0];
189      bot.events.fire(element, bot.events.EventType.CHANGE);
190      bot.events.fire(element, bot.events.EventType.BLUR);
191      return;
192    }
193  }
194
195  if (goog.isArray(values)) {
196    goog.array.forEach(values, typeValue);
197  } else {
198    typeValue(values);
199  }
200
201  if (!opt_persistModifiers) {
202    // Release all the modifier keys.
203    goog.array.forEach(bot.Keyboard.MODIFIERS, function(key) {
204      if (keyboard.isPressed(key)) {
205        keyboard.releaseKey(key);
206      }
207    });
208  }
209};

213 * Submits the form containing the given {@code element}.
214 *
215 * <p>Note this function submits the form, but does not simulate user input
216 * (a click or key press).
217 *
218 * @param {!Element} element The element to submit.

222  var form = bot.action.LegacyDevice_.findAncestorForm(element);

225                        'Element was not in a form, so could not submit.');

227  bot.action.LegacyDevice_.submitForm(element, form);

322/**
323 * Drags the given {@code element} by (dx, dy) with a virtual mouse.
324 *
325 * @param {!Element} element The element to drag.
326 * @param {number} dx Increment in x coordinate.
327 * @param {number} dy Increment in y coordinate.
328 * @param {number=} opt_steps The number of steps that should occur as part of
329 *     the drag, default is 2.
330 * @param {goog.math.Coordinate=} opt_coords Drag start position relative to the
331 *   element.
332 * @param {bot.Mouse=} opt_mouse Mouse to use; if not provided, constructs one.
333 * @throws {bot.Error} If the element cannot be interacted with.
334 */
335bot.action.drag = function(element, dx, dy, opt_steps, opt_coords, opt_mouse) {
336  var coords = bot.action.prepareToInteractWith_(element, opt_coords);
337  var initRect = bot.dom.getClientRect(element);
338  var mouse = opt_mouse || new bot.Mouse();
339  mouse.move(element, coords);
340  mouse.pressButton(bot.Mouse.Button.LEFT);
341  var steps = goog.isDef(opt_steps) ? opt_steps : 2;
342  if (steps < 1) {
343    throw new bot.Error(bot.ErrorCode.UNKNOWN_ERROR,
344                        'There must be at least one step as part of a drag.');
345  }
346  for (var i = 1; i <= steps; i++) {
347    moveTo(Math.floor(i * dx / steps), Math.floor(i * dy / steps));
348  }
349  mouse.releaseButton();
350
351  function moveTo(x, y) {
352    var currRect = bot.dom.getClientRect(element);
353    var newPos = new goog.math.Coordinate(
354        coords.x + initRect.left + x - currRect.left,
355        coords.y + initRect.top + y - currRect.top);
356    mouse.move(element, newPos);
357  }
358};

421/**
422 * Pinches the given {@code element} by the given distance with a virtual touch
423 * screen. A positive distance moves two fingers inward toward each and a
424 * negative distances spreds them outward. The optional coordinate is the point
425 * the fingers move towards (for positive distances) or away from (for negative
426 * distances); and if not provided, defaults to the center of the element.
427 *
428 * @param {!Element} element The element to pinch.
429 * @param {number} distance The distance by which to pinch the element.
430 * @param {goog.math.Coordinate=} opt_coords Position relative to the element
431 *   at the center of the pinch.
432 * @param {bot.Touchscreen=} opt_touchscreen Touchscreen to use; if not
433 *    provided, constructs one.
434 * @throws {bot.Error} If the element cannot be interacted with.
435 */
436bot.action.pinch = function(element, distance, opt_coords, opt_touchscreen) {
437  if (distance == 0) {
438    throw new bot.Error(bot.ErrorCode.UNKNOWN_ERROR,
439                        'Cannot pinch by a distance of zero.');
440  }
441  function startSoThatEndsAtMax(offsetVec) {
442    if (distance < 0) {
443      var magnitude = offsetVec.magnitude();
444      offsetVec.scale(magnitude ? (magnitude + distance) / magnitude : 0);
445    }
446  }
447  var halfDistance = distance / 2;
448  function scaleByHalfDistance(offsetVec) {
449    var magnitude = offsetVec.magnitude();
450    offsetVec.scale(magnitude ? (magnitude - halfDistance) / magnitude : 0);
451  }
452  bot.action.multiTouchAction_(element,
453                               startSoThatEndsAtMax,
454                               scaleByHalfDistance,
455                               opt_coords,
456                               opt_touchscreen);
457};

460/**
461 * Rotates the given {@code element} by the given angle with a virtual touch
462 * screen. A positive angle moves two fingers clockwise and a negative angle
463 * moves them counter-clockwise. The optional coordinate is the point to
464 * rotate around; and if not provided, defaults to the center of the element.
465 *
466 * @param {!Element} element The element to rotate.
467 * @param {number} angle The angle by which to rotate the element.
468 * @param {goog.math.Coordinate=} opt_coords Position relative to the element
469 *   at the center of the rotation.
470 * @param {bot.Touchscreen=} opt_touchscreen Touchscreen to use; if not
471 *    provided, constructs one.
472 * @throws {bot.Error} If the element cannot be interacted with.
473 */
474bot.action.rotate = function(element, angle, opt_coords, opt_touchscreen) {
475  if (angle == 0) {
476    throw new bot.Error(bot.ErrorCode.UNKNOWN_ERROR,
477                        'Cannot rotate by an angle of zero.');
478  }
479  function startHalfwayToMax(offsetVec) {
480    offsetVec.scale(0.5);
481  }
482  var halfRadians = Math.PI * (angle / 180) / 2;
483  function rotateByHalfAngle(offsetVec) {
484    offsetVec.rotate(halfRadians);
485  }
486  bot.action.multiTouchAction_(element,
487                               startHalfwayToMax,
488                               rotateByHalfAngle,
489                               opt_coords,
490                               opt_touchscreen);
491};

494/**
495 * Performs a multi-touch action with two fingers on the given element. This
496 * helper function works by manipulating an "offsetVector", which is the vector
497 * away from the center of the interaction at which the fingers are positioned.
498 * It computes the maximum offset vector and passes it to transformStart to
499 * find the starting position of the fingers; it then passes it to transformHalf
500 * twice to find the midpoint and final position of the fingers.
501 *
502 * @param {!Element} element Element to interact with.
503 * @param {function(goog.math.Vec2)} transformStart Function to transform the
504 *   maximum offset vector to the starting offset vector.
505 * @param {function(goog.math.Vec2)} transformHalf Function to transform the
506 *   offset vector halfway to its destination.
507 * @param {goog.math.Coordinate=} opt_coords Position relative to the element
508 *   at the center of the pinch.
509 * @param {bot.Touchscreen=} opt_touchscreen Touchscreen to use; if not
510 *    provided, constructs one.
511 * @private
512 */
513bot.action.multiTouchAction_ = function(element, transformStart, transformHalf,
514                                        opt_coords, opt_touchscreen) {
515  var center = bot.action.prepareToInteractWith_(element, opt_coords);
516  var size = bot.action.getInteractableSize(element);
517  var offsetVec = new goog.math.Vec2(
518      Math.min(center.x, size.width - center.x),
519      Math.min(center.y, size.height - center.y));
520
521  var touchScreen = opt_touchscreen || new bot.Touchscreen();
522  transformStart(offsetVec);
523  var start1 = goog.math.Vec2.sum(center, offsetVec);
524  var start2 = goog.math.Vec2.difference(center, offsetVec);
525  touchScreen.move(element, start1, start2);
526  touchScreen.press(/*Two Finger Press*/ true);
527
528  var initRect = bot.dom.getClientRect(element);
529  transformHalf(offsetVec);
530  var mid1 = goog.math.Vec2.sum(center, offsetVec);
531  var mid2 = goog.math.Vec2.difference(center, offsetVec);
532  touchScreen.move(element, mid1, mid2);
533
534  var midRect = bot.dom.getClientRect(element);
535  var movedVec = goog.math.Vec2.difference(
536      new goog.math.Vec2(midRect.left, midRect.top),
537      new goog.math.Vec2(initRect.left, initRect.top));
538  transformHalf(offsetVec);
539  var end1 = goog.math.Vec2.sum(center, offsetVec).subtract(movedVec);
540  var end2 = goog.math.Vec2.difference(center, offsetVec).subtract(movedVec);
541  touchScreen.move(element, end1, end2);
542  touchScreen.release();
543};

546/**
547 * Prepares to interact with the given {@code element}. It checks if the the
548 * element is shown, scrolls the element into view, and returns the coordinates
549 * of the interaction, which if not provided, is the center of the element.
550 *
551 * @param {!Element} element The element to be interacted with.
552 * @param {goog.math.Coordinate=} opt_coords Position relative to the target.
553 * @return {!goog.math.Vec2} Coordinates at the center of the interaction.
554 * @throws {bot.Error} If the element cannot be interacted with.
555 * @private
556 */
557bot.action.prepareToInteractWith_ = function(element, opt_coords) {
558  bot.action.checkShown_(element);
559  bot.action.scrollIntoView(element, opt_coords || undefined);
560
561  // NOTE: Ideally, we would check that any provided coordinates fall
562  // within the bounds of the element, but this has proven difficult, because:
563  // (1) Browsers sometimes lie about the true size of elements, e.g. when text
564  // overflows the bounding box of an element, browsers report the size of the
565  // box even though the true area that can be interacted with is larger; and
566  // (2) Elements with children styled as position:absolute will often not have
567  // a bounding box that surrounds all of their children, but it is useful for
568  // the user to be able to interact with this parent element as if it does.
569  if (opt_coords) {
570    return goog.math.Vec2.fromCoordinate(opt_coords);
571  } else {
572    var size = bot.action.getInteractableSize(element);
573    return new goog.math.Vec2(size.width / 2, size.height / 2);
574  }
575};

578/**
579 * Returns the interactable size of an element.
580 *
581 * @param {!Element} elem Element.
582 * @return {!goog.math.Size} size Size of the element.
583 */
584bot.action.getInteractableSize = function(elem) {
585  var size = goog.style.getSize(elem);
586  return ((size.width > 0 && size.height > 0) || !elem.offsetParent) ? size :
587      bot.action.getInteractableSize(elem.offsetParent);
588};

642/**
643 * Scrolls the given {@code element} in to the current viewport. Aims to do the
644 * minimum scrolling necessary, but prefers too much scrolling to too little.
645 *
646 * If an optional coordinate or rectangle region is provided, scrolls that
647 * region relative to the element into view. A coordinate is treated as a 1x1
648 * region whose top-left corner is positioned at that coordinate.
649 *
650 * @param {!Element} element The element to scroll in to view.
651 * @param {!(goog.math.Coordinate|goog.math.Rect)=} opt_region
652 *     Region relative to the top-left corner of the element.
653 * @return {boolean} Whether the element is in view after scrolling.
654 */
655bot.action.scrollIntoView = function(element, opt_region) {
656  // If the element is already in view, return true; if hidden, return false.
657  var overflow = bot.dom.getOverflowState(element, opt_region);
658  if (overflow != bot.dom.OverflowState.SCROLL) {
659    return overflow == bot.dom.OverflowState.NONE;
660  }
661
662  // Some elements may not have a scrollIntoView function - for example,
663  // elements under an SVG element. Call those only if they exist.
664  if (element.scrollIntoView) {
665    element.scrollIntoView();
666    if (bot.dom.OverflowState.NONE ==
667        bot.dom.getOverflowState(element, opt_region)) {
668      return true;
669    }
670  }
671
672  // There may have not been a scrollIntoView function, or the specified
673  // coordinate may not be in view, so scroll "manually".
674  var region = bot.dom.getClientRegion(element, opt_region);
675  for (var container = bot.dom.getParentElement(element);
676       container;
677       container = bot.dom.getParentElement(container)) {
678    scrollClientRegionIntoContainerView(container);
679  }
680  return bot.dom.OverflowState.NONE ==
681      bot.dom.getOverflowState(element, opt_region);
682
683  function scrollClientRegionIntoContainerView(container) {
684    // Based largely from goog.style.scrollIntoContainerView.
685    var containerRect = bot.dom.getClientRect(container);
686    var containerBorder = goog.style.getBorderBox(container);
687
688    // Relative position of the region to the container's content box.
689    var relX = region.left - containerRect.left - containerBorder.left;
690    var relY = region.top - containerRect.top - containerBorder.top;
691
692    // How much the region can move in the container. Use the container's
693    // clientWidth/Height, not containerRect, to account for the scrollbar.
694    var spaceX = container.clientWidth + region.left - region.right;
695    var spaceY = container.clientHeight + region.top - region.bottom;
696
697    // Scroll the element into view of the container.
698    container.scrollLeft += Math.min(relX, Math.max(relX - spaceX, 0));
699    container.scrollTop += Math.min(relY, Math.max(relY - spaceY, 0));
700  }
