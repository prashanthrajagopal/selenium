194 * @param {boolean} bubbles Whether the created events bubble.
195 * @param {boolean} cancelable Whether the created events are cancelable.

198bot.events.EventFactory_ = function(type, bubbles, cancelable) {
199  /** @private {string} */
200  this.type_ = type;
201
202  /** @private {boolean} */
203  this.bubbles_ = bubbles;
204
205  /** @private {boolean} */
206  this.cancelable_ = cancelable;

210/**
211 * Creates an event.
212 *
213 * @param {!Element|!Window} target Target element of the event.
214 * @param {bot.events.EventArgs=} opt_args Event arguments.
215 * @return {!Event} Newly created event.
216 */
217bot.events.EventFactory_.prototype.create = function(target, opt_args) {
218  var doc = goog.dom.getOwnerDocument(target);
219  var event;
220
221  if (bot.userAgent.IE_DOC_PRE9) {
222    event = doc.createEventObject();
223  } else {
224    event = doc.createEvent('HTMLEvents');
225    event.initEvent(this.type_, this.bubbles_, this.cancelable_);
226  }
227
228  return event;
229};

260/**
261 * @override
262 */
263bot.events.MouseEventFactory_.prototype.create = function(target, opt_args) {
264  // Only Gecko supports the mouse pixel scroll event.
265  if (!goog.userAgent.GECKO && this == bot.events.EventType.MOUSEPIXELSCROLL) {
266    throw new bot.Error(bot.ErrorCode.UNSUPPORTED_OPERATION,
267        'Browser does not support a mouse pixel scroll event.');
268  }
269
270  var args = /** @type {!bot.events.MouseArgs} */ (opt_args);
271  var doc = goog.dom.getOwnerDocument(target);
272  var event;
273
274  if (bot.userAgent.IE_DOC_PRE9) {
275    event = doc.createEventObject();
276    event.altKey = args.altKey;
277    event.ctrlKey = args.ctrlKey;
278    event.metaKey = args.metaKey;
279    event.shiftKey = args.shiftKey;
280    event.button = args.button;
281
282    // NOTE: ie8 does a strange thing with the coordinates passed in the event:
283    // - if offset{X,Y} coordinates are specified, they are also used for
284    //   client{X,Y}, event if client{X,Y} are also specified.
285    // - if only client{X,Y} are specified, they are also used for offset{x,y}
286    // Thus, for ie8, it is impossible to set both offset and client
287    // and have them be correct when they come out on the other side.
288    event.clientX = args.clientX;
289    event.clientY = args.clientY;
290
291    // Sets a property of the event object using Object.defineProperty.
292    // Some readonly properties of the IE event object can only be set this way.
293    var setEventProperty = function(prop, value) {
294      Object.defineProperty(event, prop, {
295        get: function() {
296          return value;
297        }
298      });
299    };
300
301    // IE has fromElement and toElement properties, no relatedTarget property.
302    // IE does not allow fromElement and toElement to be set directly, but
303    // Object.defineProperty can redefine them, when it is available. Do not
304    // use Object.defineProperties (plural) because it is even less supported.
305    // If defineProperty is unavailable, fall back to setting the relatedTarget,
306    // which many event frameworks, including jQuery and Closure, forgivingly
307    // pass on as the relatedTarget on their event object abstraction.
308    if (this == bot.events.EventType.MOUSEOUT ||
309        this == bot.events.EventType.MOUSEOVER) {
310      if (Object.defineProperty) {
311        var out = (this == bot.events.EventType.MOUSEOUT);
312        setEventProperty('fromElement', out ? target : args.relatedTarget);
313        setEventProperty('toElement', out ? args.relatedTarget : target);
314      } else {
315        event.relatedTarget = args.relatedTarget;
316      }
317    }
318
319    // IE does not allow the wheelDelta property to be set directly, so we can
320    // only do it where defineProperty is supported; otherwise store the wheel
321    // delta in the event "detail" as a last resort in case the app looks there.
322    if (this == bot.events.EventType.MOUSEWHEEL) {
323      if (Object.defineProperty) {
324        setEventProperty('wheelDelta', args.wheelDelta);
325      } else {
326        event.detail = args.wheelDelta;
327      }
328    }
329  } else {
330    var view = goog.dom.getWindow(doc);
331    event = doc.createEvent('MouseEvents');
332    var detail = 1;
333
334    // All browser but Firefox provide the wheelDelta value in the event.
335    // Firefox provides the scroll amount in the detail field, where it has the
336    // opposite polarity of the wheelDelta (upward scroll is negative) and is a
337    // factor of 40 less than the wheelDelta value. Opera provides both values.
338    // The wheelDelta value is normally some multiple of 40.
339    if (this == bot.events.EventType.MOUSEWHEEL) {
340      if (!goog.userAgent.GECKO) {
341        event.wheelDelta = args.wheelDelta;
342      }
343      if (goog.userAgent.GECKO || goog.userAgent.OPERA) {
344        detail = args.wheelDelta / -40;
345      }
346    }
347
348    // Only Gecko supports a mouse pixel scroll event, so we use it as the
349    // "standard" and pass it along as is as the "detail" of the event.
350    if (goog.userAgent.GECKO && this == bot.events.EventType.MOUSEPIXELSCROLL) {
351      detail = args.wheelDelta;
352    }
353
354    // For screenX and screenY, we set those to clientX and clientY values.
355    // While not strictly correct, applications under test depend on
356    // accurate relative positioning which is satisfied.
357    event.initMouseEvent(this.type_, this.bubbles_, this.cancelable_, view,
358        detail, /*screenX*/ args.clientX, /*screenY*/ args.clientY,
359        args.clientX, args.clientY, args.ctrlKey, args.altKey,
360        args.shiftKey, args.metaKey, args.button, args.relatedTarget);
361
362    // Trying to modify the properties throws an error,
363    // so we define getters to return the correct values.
364    if (goog.userAgent.IE &&
365        event.pageX === 0 && event.pageY === 0 && Object.defineProperty) {
366      var scrollElem = goog.dom.getDomHelper(target).getDocumentScrollElement();
367      var clientElem = goog.style.getClientViewportElement(doc);
368      var pageX = args.clientX + scrollElem.scrollLeft - clientElem.clientLeft;
369      var pageY = args.clientY + scrollElem.scrollTop - clientElem.clientTop;
370
371      Object.defineProperty(event, 'pageX', {
372        get: function() {
373          return pageX;
374        }
375      });
376      Object.defineProperty(event, 'pageY', {
377        get: function() {
378          return pageY;
379        }
380      });
381    }
382  }
383
384  return event;
385};

405/**
406 * @override
407 */
408bot.events.KeyboardEventFactory_.prototype.create = function(target, opt_args) {
409  var args = /** @type {!bot.events.KeyboardArgs} */ (opt_args);
410  var doc = goog.dom.getOwnerDocument(target);
411  var event;
412
413  if (goog.userAgent.GECKO) {
414    var view = goog.dom.getWindow(doc);
415    var keyCode = args.charCode ? 0 : args.keyCode;
416    event = doc.createEvent('KeyboardEvent');
417    event.initKeyEvent(this.type_, this.bubbles_, this.cancelable_, view,
418        args.ctrlKey, args.altKey, args.shiftKey, args.metaKey, keyCode,
419        args.charCode);
420    // https://bugzilla.mozilla.org/show_bug.cgi?id=501496
421    if (this.type_ == bot.events.EventType.KEYPRESS && args.preventDefault) {
422      event.preventDefault();
423    }
424  } else {
425    if (bot.userAgent.IE_DOC_PRE9) {
426      event = doc.createEventObject();
427    } else {  // WebKit, Opera, and IE 9+ in Standards mode.
428      event = doc.createEvent('Events');
429      event.initEvent(this.type_, this.bubbles_, this.cancelable_);
430    }
431    event.altKey = args.altKey;
432    event.ctrlKey = args.ctrlKey;
433    event.metaKey = args.metaKey;
434    event.shiftKey = args.shiftKey;
435    event.keyCode = args.charCode || args.keyCode;
436    if (goog.userAgent.WEBKIT) {
437      event.charCode = (this == bot.events.EventType.KEYPRESS) ?
438          event.keyCode : 0;
439    }
440  }
441
442  return event;
443};

463/**
464 * @override
465 */
466bot.events.TouchEventFactory_.prototype.create = function(target, opt_args) {
467  if (!bot.events.SUPPORTS_TOUCH_EVENTS) {
468    throw new bot.Error(bot.ErrorCode.UNSUPPORTED_OPERATION,
469        'Browser does not support firing touch events.');
470  }
471
472  var args = /** @type {!bot.events.TouchArgs} */ (opt_args);
473  var doc = goog.dom.getOwnerDocument(target);
474  var view = goog.dom.getWindow(doc);
475
476  // Creates a TouchList, using native touch Api, for touch events.
477  function createNativeTouchList(touchListArgs) {
478    var touches = goog.array.map(touchListArgs, function(touchArg) {
479      return doc.createTouch(view, target, touchArg.identifier,
480          touchArg.pageX, touchArg.pageY, touchArg.screenX, touchArg.screenY);
481    });
482
483    return doc.createTouchList.apply(doc, touches);
484  }
485
486  // Creates a TouchList, using simulated touch Api, for touch events.
487  function createGenericTouchList(touchListArgs) {
488    var touches = goog.array.map(touchListArgs, function(touchArg) {
489      // The target field is not part of the W3C spec, but both android and iOS
490      // add the target field to each touch.
491      return {
492        identifier: touchArg.identifier,
493        screenX: touchArg.screenX,
494        screenY: touchArg.screenY,
495        clientX: touchArg.clientX,
496        clientY: touchArg.clientY,
497        pageX: touchArg.pageX,
498        pageY: touchArg.pageY,
499        target: target
500      };
501    });
502    touches.item = function(i) {
503      return touches[i];
504    };
505    return touches;
506  }
507
508  function createTouchList(touches) {
509    return bot.events.BROKEN_TOUCH_API_ ?
510        createGenericTouchList(touches) :
511        createNativeTouchList(touches);
512  }
513
514  // As a performance optimization, reuse the created touchlist when the lists
515  // are the same, which is often the case in practice.
516  var changedTouches = createTouchList(args.changedTouches);
517  var touches = (args.touches == args.changedTouches) ?
518      changedTouches : createTouchList(args.touches);
519  var targetTouches = (args.targetTouches == args.changedTouches) ?
520      changedTouches : createTouchList(args.targetTouches);
521
522  var event;
523  if (bot.events.BROKEN_TOUCH_API_) {
524    event = doc.createEvent('MouseEvents');
525    event.initMouseEvent(this.type_, this.bubbles_, this.cancelable_, view,
526        /*detail*/ 1, /*screenX*/ 0, /*screenY*/ 0, args.clientX, args.clientY,
527        args.ctrlKey, args.altKey, args.shiftKey, args.metaKey, /*button*/ 0,
528        args.relatedTarget);
529    event.touches = touches;
530    event.targetTouches = targetTouches;
531    event.changedTouches = changedTouches;
532    event.scale = args.scale;
533    event.rotation = args.rotation;
534  } else {
535    event = doc.createEvent('TouchEvent');
536    if (goog.userAgent.product.ANDROID) {
537      // Android's initTouchEvent method is not compliant with the W3C spec.
538      event.initTouchEvent(touches, targetTouches, changedTouches,
539          this.type_, view, /*screenX*/ 0, /*screenY*/ 0, args.clientX,
540          args.clientY, args.ctrlKey, args.altKey, args.shiftKey, args.metaKey);
541    } else {
542      event.initTouchEvent(this.type_, this.bubbles_, this.cancelable_, view,
543          /*detail*/ 1, /*screenX*/ 0, /*screenY*/ 0, args.clientX,
544          args.clientY, args.ctrlKey, args.altKey, args.shiftKey, args.metaKey,
545          touches, targetTouches, changedTouches, args.scale, args.rotation);
546    }
547    event.relatedTarget = args.relatedTarget;
548  }
549
550  return event;
551};

728/**
729 * Fire a named event on a particular element.
730 *
731 * @param {!Element|!Window} target The element on which to fire the event.
732 * @param {!bot.events.EventType} type Event type.
733 * @param {bot.events.EventArgs=} opt_args Arguments to initialize the event.
734 * @return {boolean} Whether the event fired successfully or was cancelled.
735 */
736bot.events.fire = function(target, type, opt_args) {
737  var factory = /** @type {!bot.events.EventFactory_} */ (type);
738  var event = factory.create(target, opt_args);
739
740  // Ensure the event's isTrusted property is set to false, so that
741  // bot.events.isSynthetic() can identify synthetic events from native ones.
742  if (!('isTrusted' in event)) {
743    event['isTrusted'] = false;
744  }
745
746  if (bot.userAgent.IE_DOC_PRE9) {
747    return target.fireEvent('on' + factory.type_, event);
748  } else {
749    return target.dispatchEvent(event);
750  }
751};

754/**
755 * Returns whether the event was synthetically created by the atoms;
756 * if false, was created by the browser in response to a live user action.
757 *
758 * @param {!(Event|goog.events.BrowserEvent)} event An event.
759 * @return {boolean} Whether the event was synthetically created.
760 */
761bot.events.isSynthetic = function(event) {
762  var e = event.getBrowserEvent ? event.getBrowserEvent() : event;
763  return 'isTrusted' in e ? !e['isTrusted'] : false;
