27/**
28 * Returns a property, with a standardized color if it contains a
29 * convertible color.
30 * @param {string} propertyName Name of the CSS property in camelCase.
31 * @param {string} propertyValue The value of the CSS property.
32 * @return {string} The value, in a standardized format
33 *    if it is a color property.
34 */
35bot.color.standardizeColor = function(propertyName, propertyValue) {
36  if (!goog.array.contains(bot.color.COLOR_PROPERTIES_, propertyName)) {
37    return propertyValue;
38  }
39  var rgba =
40      bot.color.maybeParseRgbaColor_(propertyValue) ||
41      bot.color.maybeParseRgbColor_(propertyValue) ||
42      bot.color.maybeConvertHexOrColorName_(propertyValue);
43  return rgba ? 'rgba(' + rgba.join(', ') + ')' : propertyValue;
44};

76/**
77 * Converts a hex representation of a color to RGB.
78 * @param {string} hexOrColorName Color to convert.
79 * @return {Array} array containing [r, g, b, 1] as ints in [0, 255] or null
80 *     for invalid colors.
81 * @private
82 */
83bot.color.maybeConvertHexOrColorName_ = function(hexOrColorName) {
84  hexOrColorName = hexOrColorName.toLowerCase();
85  var hex = goog.color.names[hexOrColorName.toLowerCase()];
86  if (!hex) {
87    hex = hexOrColorName.charAt(0) == '#' ?
88        hexOrColorName : '#' + hexOrColorName;
89    if (hex.length == 4) { // of the form #RGB
90      hex = hex.replace(bot.color.HEX_TRIPLET_RE_, '#$1$1$2$2$3$3');
91    }
92
93    if (!bot.color.VALID_HEX_COLOR_RE_.test(hex)) {
94      return null;
95    }
96  }
97
98  var r = parseInt(hex.substr(1, 2), 16);
99  var g = parseInt(hex.substr(3, 2), 16);
100  var b = parseInt(hex.substr(5, 2), 16);
101
102  return [r, g, b, 1];
103};

123/**
124 * Attempts to parse a string as an rgba color.  We expect strings of the
125 * format '(r, g, b, a)', or 'rgba(r, g, b, a)', where r, g, b are ints in
126 * [0, 255] and a is a float in [0, 1].
127 * @param {string} str String to check.
128 * @return {Array.<number>} the integers [r, g, b, a] for valid colors or null
129 *     for invalid colors.
130 * @private
131 */
132bot.color.maybeParseRgbaColor_ = function(str) {
133  // Each component is separate (rather than using a repeater) so we can
134  // capture the match. Also, we explicitly set each component to be either 0,
135  // or start with a non-zero, to prevent octal numbers from slipping through.
136  var regExpResultArray = str.match(bot.color.RGBA_COLOR_RE_);
137  if (regExpResultArray) {
138    var r = Number(regExpResultArray[1]);
139    var g = Number(regExpResultArray[2]);
140    var b = Number(regExpResultArray[3]);
141    var a = Number(regExpResultArray[4]);
142    if (r >= 0 && r <= 255 &&
143        g >= 0 && g <= 255 &&
144        b >= 0 && b <= 255 &&
145        a >= 0 && a <= 1) {
146      return [r, g, b, a];
147    }
148  }
149  return null;
150};

162/**
163 * Attempts to parse a string as an rgb color.  We expect strings of the format
164 * '(r, g, b)', or 'rgb(r, g, b)', where each color component is an int in
165 * [0, 255].
166 * @param {string} str String to check.
167 * @return {Array.<number>} the integers [r, g, b, 1] for valid colors or null
168 *     for invalid colors.
169 * @private
170 */
171bot.color.maybeParseRgbColor_ = function(str) {
172  // Each component is separate (rather than using a repeater) so we can
173  // capture the match. Also, we explicitly set each component to be either 0,
174  // or start with a non-zero, to prevent octal numbers from slipping through.
175  var regExpResultArray = str.match(bot.color.RGB_COLOR_RE_);
176  if (regExpResultArray) {
177    var r = Number(regExpResultArray[1]);
178    var g = Number(regExpResultArray[2]);
179    var b = Number(regExpResultArray[3]);
180    if (r >= 0 && r <= 255 &&
181        g >= 0 && g <= 255 &&
182        b >= 0 && b <= 255) {
183      return [r, g, b, 1];
184    }
185  }
186  return null;
